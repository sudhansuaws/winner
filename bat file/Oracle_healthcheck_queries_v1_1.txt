DEFRAGMENTATION DETAILS SHEET 
--Fragmentation Query
  select owner,table_name,blocks,num_rows,avg_row_len,round(((blocks*8/1024)),2)||'MB' "TOTAL_SIZE", round((num_rows*avg_row_len/1024/1024),2)||'Mb' "ACTUAL_SIZE", round(((blocks*8/1024)-(num_rows*avg_row_len/1024/1024)),2) ||'MB' "FRAGMENTED_SPACE", round((((blocks*8/1024)-(num_rows*avg_row_len/1024/1024))/((blocks*8/1024)))*100,2) "FRAGMENTATIO_PERCENTAGE" from dba_tables where owner in(select distinct table_owner from user_indexes where table_name ='PSX_CUST_DG_BULK_TRG_T') and round(((blocks*8/1024)-(num_rows*avg_row_len/1024/1024)),2)> 100 order by 9 desc; 
--Temp Tablespace Query
  select 'Temp',sum(bytes_used/1024/1024/1024) used_gb,sum(bytes_free/1024/1024/1024) free_gb from V$temp_space_header; 
--Fragmentation Query without sys privilege
  SELECT TABLE_NAME,BLOCKS,NUM_ROWS,AVG_ROW_LEN,ROUND(((BLOCKS*8/1024)),2)||'MB' "TOTAL_SIZE", ROUND((NUM_ROWS*AVG_ROW_LEN/1024/1024),2)||'Mb' "ACTUAL_SIZE", ROUND(((BLOCKS*8/1024)-(NUM_ROWS*AVG_ROW_LEN/1024/1024)),2)"FRAGMENTED_SPACE" ,round(ROUND(((BLOCKS*8/1024)-(NUM_ROWS*AVG_ROW_LEN/1024/1024)),2)/ROUND(((BLOCKS*8/1024)),2) *100,2) fragmentated_per  FROM  USER_TABLES  WHERE ROUND(((BLOCKS*8/1024)-(NUM_ROWS*AVG_ROW_LEN/1024/1024)),2) > 100 ORDER BY 8 DESC;
